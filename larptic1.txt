    local function MorphToUser(arg_416, arg_417)
        local var_418 = Players:GetCharacterAppearanceAsync(arg_416)
        local var_419 = Players:FindFirstChild(arg_417)

        if var_419 and var_419.Character then
            local var_420, var_421, var_422 = pairs(var_419.Character:GetChildren())

                if var_423:IsA('Accessory') or (var_423:IsA('Shirt') or (var_423:IsA('Pants') or (var_423:IsA('CharacterMesh') or var_423:IsA('BodyColors')))) then

            local var_424, var_425, var_426 = pairs(var_418:GetChildren())

                    var_427.Parent = var_419.Character
                elseif var_427:IsA('Accessory') then
                    var_419.Character.Humanoid:AddAccessory(var_427)
                elseif var_427.Name == 'R15' and var_419.Character.Humanoid.RigType == Enum.HumanoidRigType.R15 then
                    local CharacterMesh = var_427:FindFirstChildOfClass('CharacterMesh')

                    if CharacterMesh then
                        CharacterMesh.Parent = var_419.Character
                    end
                end
            end

            if var_418:FindFirstChild('face') then
                var_418.face.Parent = var_419.Character.Head
            else
                local Decal = Instance.new('Decal')

                Decal.Face = Enum.NormalId.Front
                Decal.Name = 'face'
                Decal.Texture = 'rbxasset://textures/face.png'
                Decal.Transparency = 0
                Decal.Parent = var_419.Character.Head
            end

            local Parent = var_419.Character.Parent

            var_419.Character.Parent = nil
            var_419.Character.Parent = Parent
        end
    end
